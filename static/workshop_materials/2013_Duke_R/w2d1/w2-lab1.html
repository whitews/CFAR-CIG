Residual plots

# raw residuals
plot(fitted(lm.wo), resid(lm.wo), main = "Residuals vs fitted", xlab = "Fitted values", 
    ylab = "Residuals")
abline(h = 0, lty = 2)


# standardized residuals
plot(fitted(lm.wo), rstandard(lm.wo), main = "Residuals vs Fitted", xlab = "Fitted values", 
    ylab = "Standardized residuals")
abline(h = 0, lty = 2)

Normal Q-Q plots

qqnorm(rstandard(lm.wo), main = "Normal Q-Q", ylab = "Standardized residuals")
abline(a = 0, b = 1)
